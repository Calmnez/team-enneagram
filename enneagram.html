/** * 핵심동기 검사 — 유형별 게시판 백엔드 (삭제·추천 지원) * * ── 배포 방법 ── * 1) 새 구글시트를 만듭니다 (아무 이름이나 OK). * 2) 확장 프로그램 → Apps Script 를 엽니다. * 3) 기본 생성된 코드를 전부 지우고, 이 파일 내용을 통째로 붙여넣습니다. * (주의: mbti-enneagram.html 파일이 아니라 이 파일을 붙여넣어야 해요! * 이 파일은 "function doGet(e) {" 로 시작해야 정상입니다.) * 4) 저장 후 배포 → 새 배포 → 유형: '웹 앱' 선택. * - 실행 계정: '나' * - 액세스 권한: '모든 사용자' (Anyone) * 5) 배포하면 나오는 웹 앱 URL을 복사합니다. * 6) mbti-enneagram.html 파일에서 BOARD_API_URL 상수에 그 URL을 붙여넣습니다. * * ── 시트 구조 ── * 'Messages' 시트에 [Timestamp, Type, Nickname, Message, Token, Likes] 6개 열이 자동 생성됩니다. * - Token: 글쓴이 브라우저에만 저장되는 삭제용 비밀값 (응답에는 포함 안 됨) * - Likes: 추천 수 (정수) * 부적절한 메시지는 이 시트에서 해당 행을 직접 삭제하면 됩니다 (수동 모더레이션). */ const SHEET_NAME = 'Messages'; const MAX_MESSAGES_PER_TYPE = 50; // 유형별로 최근 N개만 응답에 포함 function getOrCreateSheet_() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(['Timestamp', 'Type', 'Nickname', 'Message', 'Token', 'Likes']); } return sheet; } function jsonOut_(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function timestampToIso_(v) { return v instanceof Date ? v.toISOString() : String(v); } function findRowByTimestamp_(sheet, timestamp) { const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (timestampToIso_(data[i][0]) === timestamp) return i + 1; // 1-based, 헤더 포함 } return -1; } function doGet(e) { const sheet = getOrCreateSheet_(); const type = String((e.parameter && e.parameter.type) || ''); const data = sheet.getDataRange().getValues(); const rows = data.slice(1); const filtered = rows .filter(function (r) { return String(r[1]) === type; }) .slice(-MAX_MESSAGES_PER_TYPE) .reverse(); // 최신순 const messages = filtered.map(function (r) { return { timestamp: timestampToIso_(r[0]), type: r[1], nickname: r[2], message: r[3], likes: Number(r[5]) || 0 // r[4](Token)는 응답에 포함하지 않음 — 삭제 검증용으로만 서버에 보관 }; }); return jsonOut_({ success: true, messages: messages }); } function doPost(e) { try { const sheet = getOrCreateSheet_(); const body = JSON.parse(e.postData.contents); const action = body.action || 'add'; if (action === 'add') { const type = String(body.type || '').slice(0, 4); const nickname = String(body.nickname || '익명').slice(0, 20); const message = String(body.message || '').trim().slice(0, 200); const token = String(body.token || ''); if (!type || !message) { return jsonOut_({ success: false, error: 'missing type or message' }); } const now = new Date(); sheet.appendRow([now, type, nickname, message, token, 0]); return jsonOut_({ success: true, timestamp: now.toISOString() }); } if (action === 'delete') { const timestamp = String(body.timestamp || ''); const token = String(body.token || ''); const row = findRowByTimestamp_(sheet, timestamp); if (row < 0) return jsonOut_({ success: false, error: 'not found' }); const storedToken = String(sheet.getRange(row, 5).getValue() || ''); if (!token || token !== storedToken) { return jsonOut_({ success: false, error: 'token mismatch' }); } sheet.deleteRow(row); return jsonOut_({ success: true }); } if (action === 'like' || action === 'unlike') { const timestamp = String(body.timestamp || ''); const row = findRowByTimestamp_(sheet, timestamp); if (row < 0) return jsonOut_({ success: false, error: 'not found' }); const cell = sheet.getRange(row, 6); let likes = Number(cell.getValue()) || 0; likes = action === 'like' ? likes + 1 : Math.max(0, likes - 1); cell.setValue(likes); return jsonOut_({ success: true, likes: likes }); } return jsonOut_({ success: false, error: 'unknown action: ' + action }); } catch (err) { return jsonOut_({ success: false, error: String(err) }); } }
